基础样例 #

设置图标 #




<template>
	<!-- 字符串格式 -->
	<o-icon icon="mask" type="fad" size="2x"></o-icon>
	<o-icon icon="mask" type="fas" size="2x"></o-icon>
	<br />
	<!-- 数组格式 -->
	<o-icon :icon="['fad', 'mask']" size="2x"></o-icon>
	<o-icon :icon="['fas', 'mask']" type="fas" size="2x"></o-icon>
	<br />
	<!-- 对象格式 -->
	<o-icon ref="icon" :icon="computedMask" size="2x"></o-icon>
	<o-icon :icon="fadMask" size="2x"></o-icon>
	<o-icon :icon="fasMask" size="2x"></o-icon>
</template>
<script lang="ts">
import { faMask as fadMask } from '@fortawesome/pro-duotone-svg-icons';
import { faMask as fasMask } from '@fortawesome/pro-solid-svg-icons';
import { OIcon } from '@owl/skeleton';
import { addIcon } from '@yiird/owl';
import { computed, defineComponent, ref } from 'vue';

addIcon(fadMask, fasMask);
export default defineComponent({
	name: 'IconNameDemo',
	setup() {
		const icon = ref<InstanceType<typeof OIcon>>();

		const computedMask = computed(() => {
			return fadMask;
		});

		const doClick = () => {
			alert(11);
		};

		return {
			computedMask,
			fadMask,
			fasMask,
			icon,
			doClick
		};
	}
});
</script>

<template>
	<div>
		<o-button @click="changeSize('xs')">xs</o-button>
		<o-button @click="changeSize('sm')">sm</o-button>
		<o-button @click="changeSize('lg')">lg</o-button>
		<o-button @click="changeSize('1x')">1x</o-button>
		<o-button @click="changeSize('2x')">2x</o-button>
		<o-button @click="changeSize('3x')">3x</o-button>
		<o-button @click="changeSize('4x')">4x</o-button>
		<o-button @click="changeSize('5x')">5x</o-button>
		<o-button @click="changeSize('6x')">6x</o-button>
		<o-button @click="changeSize('7x')">7x</o-button>
		<o-button @click="changeSize('8x')">8x</o-button>
		<o-button @click="changeSize('9x')">9x</o-button>
		<o-button @click="changeSize('10x')">10x</o-button>
	</div>
	<div>
		<o-icon icon="heart" :size="sizeRef"></o-icon>
	</div>
</template>
<script lang="ts">
import { faHeart } from '@fortawesome/pro-solid-svg-icons';
import { addIcon } from '@yiird/owl';
import { defineComponent, ref } from 'vue';
addIcon(faHeart);
export default defineComponent({
	name: 'IconSizeDemo',
	setup() {
		const sizeRef = ref('xs');

		const changeSize = (size: string) => {
			sizeRef.value = size;
		};
		return {
			sizeRef,
			changeSize
		};
	}
});
</script>

固定宽度 #


需要垂直对齐一系列图标，比如在列表或导航菜单中？ 我们为这些情况提供了固定宽度的支持样式。
未设置fixed-width
设置fixed-width
<template>
	<div style="clear: both; overflow: hidden">
		<div style="float: left; overflow: hidden; width: 200px">
			<div><o-icon icon="skating" size="5x" style="background: DodgerBlue"></o-icon></div>
			<div><o-icon icon="skiing" size="5x" style="background: skyblue"></o-icon></div>
			<div><o-icon icon="skiing-nordic" size="5x" style="background: DodgerBlue"></o-icon></div>
			<div><o-icon icon="snowboarding" size="5x" style="background: skyblue"></o-icon></div>
			<div><o-icon icon="snowplow" size="5x" style="background: DodgerBlue"></o-icon></div>
			<span>未设置fixed-width</span>
		</div>
		<div style="float: left; overflow: hidden; width: 200px">
			<div><o-icon icon="skating" fixed-width size="5x" style="background: DodgerBlue"></o-icon></div>
			<div><o-icon icon="skiing" fixed-width size="5x" style="background: skyblue"></o-icon></div>
			<div><o-icon icon="skiing-nordic" fixed-width size="5x" style="background: DodgerBlue"></o-icon></div>
			<div><o-icon icon="snowboarding" fixed-width size="5x" style="background: skyblue"></o-icon></div>
			<div><o-icon icon="snowplow" fixed-width size="5x" style="background: DodgerBlue"></o-icon></div>
			<span>设置fixed-width</span>
		</div>
	</div>
</template>
<script lang="ts">
import { faSkating, faSkiing, faSkiingNordic, faSnowboarding, faSnowplow } from '@fortawesome/pro-solid-svg-icons';
import { addIcon } from '@yiird/owl';
import { defineComponent } from 'vue';

addIcon(faSkating, faSkiing, faSkiingNordic, faSnowboarding, faSnowplow);
export default defineComponent({
	name: 'IconFixedWidthDemo'
});
</script>
